\section{Lineare Abbildungen}

\subsection{Definition}
Eine Abbildung $f: V \to W$ zwischen zwei reellen Vektorräumen heisst \emph{linear}, wenn für alle $\vec{x}, \vec{y} \in V$ und $\lambda \in \mathbb{R}$ gilt:
\begin{enumerate}
    \item $f(\vec{x} + \vec{y}) = f(\vec{x}) + f(\vec{y})$ \quad (Additivität)
    \item $f(\lambda \cdot \vec{x}) = \lambda \cdot f(\vec{x})$ \quad (Homogenität)
\end{enumerate}
$f(\vec{x}) \in W$ heisst \emph{Bild} von $\vec{x}$.

\textbf{Nicht linear} sind z.B.: $f(x) = x + 5$, $f(x) = x^2$, $f(x) = \cos(x)$, $f(x) = e^x$.\\
Die einzigen linearen Abbildungen $\mathbb{R} \to \mathbb{R}$ sind $f(x) = m \cdot x$.

\subsection{Abbildungsmatrix (Standardbasis)}
Jede lineare Abbildung $f: \mathbb{R}^n \to \mathbb{R}^m$ lässt sich durch eine $m \times n$-Matrix $A$ darstellen:
$$f(\vec{x}) = A \cdot \vec{x}$$
Die Spalten von $A$ sind die Bilder der Standardbasisvektoren:
$$A = \begin{pmatrix} | & | & & | \\ f(\vec{e}_1) & f(\vec{e}_2) & \cdots & f(\vec{e}_n) \\ | & | & & | \end{pmatrix}$$

\subsection{Abbildungsmatrix (beliebige Basen)}
Für Vektorräume $V$ mit Basis $\mathcal{B} = \{\vec{b}_1, \ldots, \vec{b}_n\}$ und $W$ mit Basis $\mathcal{C} = \{\vec{c}_1, \ldots, \vec{c}_m\}$:
$$(f(\vec{x}))_\mathcal{C} = {}_\mathcal{C}A_\mathcal{B} \cdot \vec{x}_\mathcal{B}$$
$${}_\mathcal{C}A_\mathcal{B} = \begin{pmatrix} | & | & & | \\ (f(\vec{b}_1))_\mathcal{C} & (f(\vec{b}_2))_\mathcal{C} & \cdots & (f(\vec{b}_n))_\mathcal{C} \\ | & | & & | \end{pmatrix}$$
Die Spalten erhält man durch Lösen von LGS: $f(\vec{b}_i)$ als Linearkombination der $\vec{c}_j$ darstellen.

\subsection{Wichtige lineare Abbildungen $\mathbb{R}^2 \to \mathbb{R}^2$}

\begin{tabular}{@{}ll@{}}
\textbf{Streckung} um $\lambda_1$ in $x$, $\lambda_2$ in $y$ & $\begin{pmatrix} \lambda_1 & 0 \\ 0 & \lambda_2 \end{pmatrix}$ \\[8pt]
\textbf{Rotation} um Winkel $\varphi$ & $\begin{pmatrix} \cos\varphi & -\sin\varphi \\ \sin\varphi & \cos\varphi \end{pmatrix}$ \\[8pt]
\textbf{Spiegelung} an $x$-Achse & $\begin{pmatrix} 1 & 0 \\ 0 & -1 \end{pmatrix}$ \\[8pt]
\textbf{Spiegelung} an $y$-Achse & $\begin{pmatrix} -1 & 0 \\ 0 & 1 \end{pmatrix}$ \\[8pt]
\textbf{Punktspiegelung} am Ursprung & $\begin{pmatrix} -1 & 0 \\ 0 & -1 \end{pmatrix}$ \\[8pt]
\textbf{Scherung} in $x$-Ri.\ mit Faktor $m$ & $\begin{pmatrix} 1 & m \\ 0 & 1 \end{pmatrix}$ \\[8pt]
\textbf{Proj.\ auf} $x$-Achse & $\begin{pmatrix} 1 & 0 \\ 0 & 0 \end{pmatrix}$ \\[8pt]
\textbf{Proj.\ auf} $y$-Achse & $\begin{pmatrix} 0 & 0 \\ 0 & 1 \end{pmatrix}$
\end{tabular}

\subsection{Projektion \& Spiegelung an Gerade}
Für Gerade $g: ax + by = 0$ mit $a^2 + b^2 = 1$ (normiert!):

\textbf{Orthogonale Projektion:}
$P = \begin{pmatrix} 1-a^2 & -ab \\ -ab & 1-b^2 \end{pmatrix}$, \quad $p(\vec{x}) = \vec{x} - (\vec{n} \cdot \vec{x}) \cdot \vec{n}$

\textbf{Spiegelung:}
$S = \begin{pmatrix} 1-2a^2 & -2ab \\ -2ab & 1-2b^2 \end{pmatrix}$, \quad $s(\vec{x}) = \vec{x} - 2(\vec{n} \cdot \vec{x}) \cdot \vec{n}$

Dabei ist $\vec{n} = \binom{a}{b}$ der normierte Normalenvektor.

\subsection{Lineare Abbildungen $\mathbb{R}^3 \to \mathbb{R}^3$}

\textbf{Zentrische Streckung} um Faktor $\lambda$: $A = \begin{pmatrix} \lambda & 0 & 0 \\ 0 & \lambda & 0 \\ 0 & 0 & \lambda \end{pmatrix}$

\begin{tabular}{@{}ll@{}}
\textbf{Proj.\ auf} $xy$-Ebene & $\begin{pmatrix} 1&0&0\\0&1&0\\0&0&0 \end{pmatrix}$ \\[6pt]
\textbf{Spieg.\ an} $xy$-Ebene & $\begin{pmatrix} 1&0&0\\0&1&0\\0&0&-1 \end{pmatrix}$ \\[6pt]
\textbf{Proj.\ auf} $x$-Achse & $\begin{pmatrix} 1&0&0\\0&0&0\\0&0&0 \end{pmatrix}$ \\[6pt]
\textbf{Spieg.\ an} $x$-Achse & $\begin{pmatrix} 1&0&0\\0&-1&0\\0&0&-1 \end{pmatrix}$
\end{tabular}

Analog für $xz$-/$yz$-Ebene und $y$-/$z$-Achse durch Vertauschen der Einträge.

\subsection{Projektion \& Spiegelung an Ebene ($\mathbb{R}^3$)}
Für Ebene $E: ax + by + cz = 0$ mit $a^2+b^2+c^2=1$, $\vec{n}=\begin{pmatrix}a\\b\\c\end{pmatrix}$:

\textbf{Orth.\ Projektion:} $P = E - \vec{n}\cdot\vec{n}^T = \begin{pmatrix} 1-a^2 & -ab & -ac \\ -ab & 1-b^2 & -bc \\ -ac & -bc & 1-c^2 \end{pmatrix}$

\textbf{Spiegelung:} $S = E - 2\vec{n}\cdot\vec{n}^T = \begin{pmatrix} 1-2a^2 & -2ab & -2ac \\ -2ab & 1-2b^2 & -2bc \\ -2ac & -2bc & 1-2c^2 \end{pmatrix}$

\subsection{Rotationen in $\mathbb{R}^3$}

\begin{tabular}{@{}ll@{}}
    \textbf{Um $x$-Achse} & $\begin{pmatrix} 1 & 0 & 0 \\ 0 & \cos\varphi & -\sin\varphi \\ 0 & \sin\varphi & \cos\varphi \end{pmatrix}$ \\[10pt]
    \textbf{Um $y$-Achse} & $\begin{pmatrix} \cos\varphi & 0 & \sin\varphi \\ 0 & 1 & 0 \\ -\sin\varphi & 0 & \cos\varphi \end{pmatrix}$ \\[10pt]
    \textbf{Um $z$-Achse} & $\begin{pmatrix} \cos\varphi & -\sin\varphi & 0 \\ \sin\varphi & \cos\varphi & 0 \\ 0 & 0 & 1 \end{pmatrix}$
\end{tabular}

\textbf{Rotation um allg.\ Achse} durch Ursprung mit norm.\ Vektor $\vec{a}=\begin{pmatrix}a_1\\a_2\\a_3\end{pmatrix}$:
{\tiny
$R = \begin{pmatrix} c+a_1^2(1-c) & a_1a_2(1-c)-a_3s & a_1a_3(1-c)+a_2s \\ a_1a_2(1-c)+a_3s & c+a_2^2(1-c) & a_2a_3(1-c)-a_1s \\ a_1a_3(1-c)-a_2s & a_2a_3(1-c)+a_1s & c+a_3^2(1-c) \end{pmatrix}$
}

mit $c = \cos(\varphi)$, $s = \sin(\varphi)$.

\subsection{Kern und Bild}

\subsubsection*{Kern}
Der \textbf{Kern} einer linearen Abbildung $f: V \to W$ ist die Menge aller Vektoren aus $V$, die durch $f$ auf den Nullvektor $\vec{0}$ in $W$ abgebildet werden. \\[6pt]

\textbf{Definition:} $\ker(f) = \{\vec{x} \in V \mid f(\vec{x}) = \vec{0}\}$

Für die Abbildungsmatrix $A$ (Dimension $m \times n$):
$$\ker(A) = \{\vec{x} \in \mathbb{R}^n \mid A \cdot \vec{x} = \vec{0}\}$$

Dies ist die \textbf{Lösungsmenge des homogenen linearen Gleichungssystems} $A \vec{x} = \vec{0}$.

\textbf{Eigenschaften:}
\begin{itemize}
    \item $\ker(f)$ ist ein Unterraum von $V$
    \item $\vec{0} \in \ker(f)$ immer (da $f(\vec{0}) = \vec{0}$)
    \item $f$ ist injektiv $\Leftrightarrow$ $\ker(f) = \{\vec{0}\}$ (nur der Nullvektor im Kern)
\end{itemize}

\subsubsection*{Bild}
Das \textbf{Bild} einer linearen Abbildung $f: V \to W$ ist die Menge aller Vektoren aus $W$, die als Bild von mindestens einem Vektor aus $V$ auftreten. \\[6pt]

\textbf{Definition:} $\text{im}(f) = \{f(\vec{x}) \mid \vec{x} \in V\} = \{\vec{y} \in W \mid \exists \vec{x} \in V: f(\vec{x}) = \vec{y}\}$

Für die Abbildungsmatrix $A$ (Dimension $m \times n$):
$$\text{im}(A) = \text{span}(\vec{a}_1, \vec{a}_2, \ldots, \vec{a}_n)$$

Das Bild wird \textbf{aufgespannt} von den Spalten von $A$ (aufgespannter Spaltenraum). \\[6pt]

\textbf{Eigenschaften:}
\begin{itemize}
    \item $\text{im}(f)$ ist ein Unterraum von $W$
    \item $\text{im}(f)$ ist der Spaltenraum von $A$
    \item $\dim(\text{im}(A)) = \text{Rang}(A)$ = Anzahl linear unabhängiger Spalten
    \item $f$ ist surjektiv $\Leftrightarrow$ $\text{im}(f) = W$
\end{itemize}

\subsubsection*{Dimensionssatz}
Für eine lineare Abbildung $f: V \to W$ mit $\dim(V) = n$ gilt:
$$\dim(\ker(f)) + \dim(\text{im}(f)) = n$$

Äquivalent mit Abbildungsmatrix $A$ ($m \times n$):
$$\dim(\text{im}(A)) = \text{rg}(A)$$


\subsection{Verknüpfung linearer Abbildungen}
$f: U \to V$ mit Matrix $A$, $g: V \to W$ mit Matrix $B$:
$$g \circ f: \vec{x} \mapsto B \cdot (A \cdot \vec{x}) = (B \cdot A) \cdot \vec{x}$$
Die Matrix der \textbf{zuerst} ausgeführten Abbildung steht \textbf{rechts} im Produkt.

\textbf{Spiegelungen:} $S^2 = E$ \\
\textbf{Projektionen:} $P^2 = P$

\subsection{Inverse einer linearen Abbildung}
$f$ invertierbar $\Leftrightarrow$ $A$ invertierbar. Die Abbildungsmatrix der Inversen von $f$ ist $A^{-1}$.\\
Es gilt: $B \cdot A = E \Rightarrow B = A^{-1}$.\\
Projektionen sind \textbf{nicht} invertierbar ($\det(P) = 0$).

\subsection{Basiswechsel}
$\mathcal{S}$: Standardbasis, $\mathcal{B}$: andere Basis mit Vektoren $\vec{b}_1, \ldots, \vec{b}_n$.

\textbf{$\mathcal{B} \to \mathcal{S}$:} ${}_\mathcal{S}T_\mathcal{B} = \begin{pmatrix} | & | \\ \vec{b}_{1\mathcal{S}} & \vec{b}_{2\mathcal{S}} \\ | & | \end{pmatrix}$ (Spalten = Basisvektoren von $\mathcal{B}$ bzgl.\ $\mathcal{S}$)

\textbf{$\mathcal{S} \to \mathcal{B}$:} ${}_\mathcal{B}T_\mathcal{S} = {}_\mathcal{S}T_\mathcal{B}^{-1}$

\textbf{Zusammenhang:} ${}_\mathcal{C}A_\mathcal{C} = {}_\mathcal{C}T_\mathcal{B} \cdot {}_\mathcal{B}A_\mathcal{B} \cdot {}_\mathcal{B}T_\mathcal{C} = {}_\mathcal{C}T_\mathcal{B} \cdot {}_\mathcal{B}A_\mathcal{B} \cdot {}_\mathcal{C}T_\mathcal{B}^{-1}$

\subsection{Homogene Koordinaten}
Translationen sind \textbf{nicht linear}. Trick: Vektoren um eine Komponente erweitern.
\begin{itemize}
    \item \textbf{Ortsvektoren} (Punkt) sind am Ursprung angeheftet; letzte Komponente $= 1$ 
    \item \textbf{Freier Vektoren} sind frei parallel verschiebbar; letzte Komponente $= 0$
\end{itemize}
\begin{align*}
    \text{Ortsvektor: } \vec{p} = \begin{pmatrix} x_1 \\ x_2 \\ \vdots \\ x_n \\ 1 \end{pmatrix} \quad
    \text{Freier Vektor: } \vec{v} = \begin{pmatrix} v_1 \\ v_2 \\ \vdots \\ v_n \\ 0 \end{pmatrix}
\end{align*}

\subsubsection{Homogene Abbildungsmatrizen}
\textbf{Translation} in $\mathbb{R}^2$ um $\vec{a}=\binom{a_1}{a_2}$: $T = \begin{pmatrix} 1 & 0 & a_1 \\ 0 & 1 & a_2 \\ 0 & 0 & 1 \end{pmatrix}$

\textbf{Rotation + Translation} in $\mathbb{R}^2$:
$\begin{pmatrix} \cos\varphi & -\sin\varphi & a_1 \\ \sin\varphi & \cos\varphi & a_2 \\ 0 & 0 & 1 \end{pmatrix}$

\textbf{Translation} in $\mathbb{R}^3$ um $\vec{a}=\begin{pmatrix}a_1\\a_2\\a_3\end{pmatrix}$: $T = \begin{pmatrix} 1&0&0&a_1 \\ 0&1&0&a_2 \\ 0&0&1&a_3 \\ 0&0&0&1 \end{pmatrix}$

Lineare Abbildungsmatrizen werden analog um Zeile/Spalte mit $(0\;\cdots\;0\;1)$ erweitert.
